# Semantic Photo Search + Browse UI — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Browser UI where the user types a text query or clicks a photo to find visually similar photos, served locally via FastAPI on port 8890.

**Architecture:** FastAPI app at `scripts/search_ui/app.py` loads the SigLIP text encoder once at startup, encodes text queries to 1152-dim vectors, and issues cosine ANN queries against `photo_embeddings` via pgvector. The single-page frontend (`ui/search.html`) hits the API for text search, similar-image search, and tag browse; photos are served as on-the-fly 800px thumbnails.

**Tech Stack:** FastAPI, uvicorn, psycopg[binary], pgvector, Pillow, transformers (SigLIP text tower only), torch (CPU), httpx (tests), pytest, Docker (host networking), vanilla JS (no build step).

  ```python
  from __future__ import annotations

  import numpy as np
  import psycopg

  MODEL_NAME = "siglip-so400m"


  def ann_search(
      conn: psycopg.Connection,
      query_vec: np.ndarray,
      *,
      k: int = 50,
      media_type: str = "photo",
  ) -> list[tuple[str, float, list[str]]]:
      """Return top-k (sha256, score, tags) by cosine similarity.

      score = 1 - cosine_distance, so higher is more similar.
      Tags are fetched in the same query via a LEFT JOIN on photo_tags.
      """
      vec_list = query_vec.tolist()
      sql = """
          WITH ranked AS (
              SELECT pe.sha256,
                     1 - (pe.embedding <=> %s::vector) AS score
              FROM photo_embeddings pe
              WHERE pe.model = %s
                AND pe.media_type = %s
              ORDER BY pe.embedding <=> %s::vector
              LIMIT %s
          )
          SELECT r.sha256,
                 r.score,
                 COALESCE(array_agg(pt.label ORDER BY pt.score DESC)
                          FILTER (WHERE pt.label IS NOT NULL), ARRAY[]::text[]) AS tags
          FROM ranked r
          LEFT JOIN photo_tags pt ON pt.sha256 = r.sha256
          GROUP BY r.sha256, r.score
          ORDER BY r.score DESC
      """
      with conn.cursor() as cur:
          cur.execute(sql, (vec_list, MODEL_NAME, media_type, vec_list, k))
          rows = cur.fetchall()
      return [(sha, float(score), list(tags)) for sha, score, tags in rows]


  def fetch_tags(conn: psycopg.Connection) -> list[dict]:
      """Return all distinct tag labels with photo counts, sorted by count desc."""
      sql = """
          SELECT label, COUNT(DISTINCT sha256) AS cnt
          FROM photo_tags
          GROUP BY label
          ORDER BY cnt DESC
      """
      with conn.cursor() as cur:
          cur.execute(sql)
          rows = cur.fetchall()
      return [{"label": label, "count": int(cnt)} for label, cnt in rows]


  def fetch_by_tag(
      conn: psycopg.Connection,
      label: str,
      *,
      k: int = 50,
  ) -> list[tuple[str, float]]:
      """Return (sha256, score) for photos with the given tag, sorted by score."""
      sql = """
          SELECT sha256, score
          FROM photo_tags
          WHERE label = %s
          ORDER BY score DESC
          LIMIT %s
      """
      with conn.cursor() as cur:
          cur.execute(sql, (label, k))
          rows = cur.fetchall()
      return [(sha, float(score)) for sha, score in rows]


  def fetch_embedding(
      conn: psycopg.Connection,
      sha256: str,
      *,
      media_type: str = "photo",
  ) -> np.ndarray | None:
      """Fetch the stored embedding vector for a given sha256, or None if missing."""
      sql = """
          SELECT embedding
          FROM photo_embeddings
          WHERE sha256 = %s AND model = %s AND media_type = %s
          LIMIT 1
      """
      with conn.cursor() as cur:
          cur.execute(sql, (sha256, MODEL_NAME, media_type))
          row = cur.fetchone()
      if row is None:
          return None
      return np.array(row[0], dtype=np.float32)
  ```

  ```python
  from __future__ import annotations

  import numpy as np
  import torch
  from transformers import AutoModel, AutoProcessor

  MODEL_HF_ID = "google/siglip-so400m-patch14-384"


  class SigLIPTextEncoder:
      """Thin wrapper around the SigLIP text tower.

      Loaded once at startup; encode() is ~100 ms on CPU.
      """

      def __init__(self, model_id: str = MODEL_HF_ID) -> None:
          self._processor = AutoProcessor.from_pretrained(model_id)
          self._model = AutoModel.from_pretrained(model_id)
          self._model.eval()

      def encode(self, text: str) -> np.ndarray:
          """Return a normalised float32 vector of shape (1152,)."""
          inputs = self._processor(
              text=[text],
              return_tensors="pt",
              padding="max_length",
              truncation=True,
          )
          with torch.no_grad():
              out = self._model.get_text_features(**inputs)
          # SigLIP returns SigLIPTextModelOutput with pooler_output; fall back
          # to raw tensor when pooler_output is absent (e.g. in tests).
          if hasattr(out, "pooler_output"):
              features = out.pooler_output
          else:
              features = out
          features = features / features.norm(dim=-1, keepdim=True)
          return features.squeeze(0).cpu().numpy().astype(np.float32)
  ```

  ```python
  from __future__ import annotations

  import io
  import os
  from contextlib import asynccontextmanager
  from pathlib import Path

  import numpy as np
  import psycopg
  from fastapi import FastAPI, HTTPException, Query
  from fastapi.responses import HTMLResponse, Response
  from pgvector.psycopg import register_vector
  from PIL import Image

  from search_ui.db import ann_search, fetch_by_tag, fetch_embedding, fetch_tags
  from search_ui.model import SigLIPTextEncoder

  # ---------------------------------------------------------------------------
  # Configuration
  # ---------------------------------------------------------------------------
  DATABASE_URL = os.environ.get(
      "DATABASE_URL",
      "postgresql://dedup:dedup_local_dev@localhost:5432/dedup",
  )
  MEDIA_DIR = Path(os.environ.get("MEDIA_DIR", "/home/will/photo_project/originals/media"))
  SEARCH_UI_DIR = Path(__file__).resolve().parents[2] / "ui"
  THUMB_MAX = 800

  # Set at startup via lifespan; patched to mocks in tests
  TEXT_ENCODER: SigLIPTextEncoder | None = None
  _DB_CONN: psycopg.Connection | None = None


  def get_db_conn() -> psycopg.Connection:
      return _DB_CONN  # type: ignore[return-value]


  # ---------------------------------------------------------------------------
  # Lifespan: load model + open DB connection once
  # ---------------------------------------------------------------------------
  @asynccontextmanager
  async def lifespan(app: FastAPI):
      global TEXT_ENCODER, _DB_CONN
      TEXT_ENCODER = SigLIPTextEncoder()
      _DB_CONN = psycopg.connect(DATABASE_URL)
      register_vector(_DB_CONN)
      yield
      if _DB_CONN:
          _DB_CONN.close()


  app = FastAPI(title="Photo Search", lifespan=lifespan)


  # ---------------------------------------------------------------------------
  # Helpers
  # ---------------------------------------------------------------------------
  def _thumb_bytes(path: Path) -> bytes:
      img = Image.open(path).convert("RGB")
      if max(img.size) > THUMB_MAX:
          img.thumbnail((THUMB_MAX, THUMB_MAX), Image.LANCZOS)
      buf = io.BytesIO()
      img.save(buf, format="JPEG", quality=85)
      return buf.getvalue()


  def _find_photo(sha256: str) -> Path:
      for ext in (".jpg", ".jpeg", ".png", ".gif", ".bmp", ".tif", ".tiff", ".heic"):
          p = MEDIA_DIR / f"{sha256}{ext}"
          if p.exists():
              return p
      raise HTTPException(status_code=404, detail=f"Photo {sha256} not found")


  def _sha_param(sha256: str) -> str:
      if len(sha256) != 64:
          raise HTTPException(status_code=422, detail="sha256 must be 64 hex chars")
      return sha256


  def _format_results(rows) -> list[dict]:
      return [{"sha256": sha, "score": round(score, 6), "tags": tags}
              for sha, score, tags in rows]


  def _format_tag_results(rows) -> list[dict]:
      return [{"sha256": sha, "score": round(score, 6), "tags": []}
              for sha, score in rows]


  # ---------------------------------------------------------------------------
  # Routes
  # ---------------------------------------------------------------------------
  @app.get("/", response_class=HTMLResponse)
  async def root():
      html_path = SEARCH_UI_DIR / "search.html"
      if not html_path.exists():
          return HTMLResponse("<h1>search.html not found</h1>", status_code=200)
      return HTMLResponse(html_path.read_text())


  @app.get("/photo/{sha256}")
  async def serve_photo(sha256: str):
      sha256 = _sha_param(sha256)
      path = _find_photo(sha256)
      data = _thumb_bytes(path)
      return Response(content=data, media_type="image/jpeg")


  @app.get("/search")
  async def text_search(q: str = Query(..., min_length=1), k: int = Query(50, ge=1, le=200)):
      vec = TEXT_ENCODER.encode(q)
      conn = get_db_conn()
      rows = ann_search(conn, vec, k=k)
      return _format_results(rows)


  @app.get("/similar/{sha256}")
  async def similar_search(sha256: str, k: int = Query(50, ge=1, le=200)):
      sha256 = _sha_param(sha256)
      conn = get_db_conn()
      vec = fetch_embedding(conn, sha256)
      if vec is None:
          raise HTTPException(status_code=404, detail=f"No embedding for {sha256}")
      rows = ann_search(conn, vec, k=k)
      return _format_results(rows)


  @app.get("/tags")
  async def tags():
      conn = get_db_conn()
      return fetch_tags(conn)


  @app.get("/by-tag")
  async def by_tag(label: str = Query(..., min_length=1), k: int = Query(50, ge=1, le=200)):
      conn = get_db_conn()
      rows = fetch_by_tag(conn, label, k=k)
      return _format_tag_results(rows)
  ```
